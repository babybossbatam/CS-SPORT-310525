Set Up the Backend Proxy
Create a new folder for your backend (e.g., football-backend).
Open a terminal in that folder and run:
bash
Copy Code
npm init -y
npm install express node-fetch cors
Create a file named server.js and paste this code (replace YOUR_API_KEY with your real RapidAPI key):
js
Copy Code
const express = require('express');
const fetch = require('node-fetch');
const cors = require('cors');

const app = express();
app.use(cors()); // Allow all origins (for dev)

const API_URL = 'https://api-football-v1.p.rapidapi.com/v3/players/topscorers?league=39&season=2025';
const API_KEY = 'YOUR_API_KEY'; // <-- Replace with your real API key

app.get('/api/topscorers', async (req, res) => {
  try {
    const response = await fetch(API_URL, {
      method: 'GET',
      headers: {
        'x-rapidapi-key': API_KEY,
        'x-rapidapi-host': 'api-football-v1.p.rapidapi.com'
      }
    });
    const data = await response.json();
    res.json(data);
  } catch (err) {
    res.status(500).json({ error: 'Failed to fetch data' });
  }
});

const PORT = 5000;
app.listen(PORT, () => console.log(`Proxy server running on port ${PORT}`));
Start your backend server:
bash
Copy Code
node server.js
You should see:
Proxy server running on port 5000

import React, { useEffect, useState } from "react";

function TopScorers() {
  const [players, setPlayers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch("http://localhost:5000/api/topscorers")
      .then((res) => res.json())
      .then((data) => {
        setPlayers(data.response || []);
        setLoading(false);
      })
      .catch((err) => {
        setError("Failed to fetch data from proxy.");
        setLoading(false);
      });
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div style={{ color: 'red' }}>{error}</div>;

  return (
    <div style={{ display: "flex", flexWrap: "wrap", gap: "16px" }}>
      {players.map((playerObj) => (
        <div
          key={playerObj.player.id}
          style={{
            border: "1px solid #ccc",
            borderRadius: "8px",
            padding: "8px",
            width: "150px",
            textAlign: "center",
          }}
        >
          <img
            src={playerObj.player.photo}
            alt={playerObj.player.name}
            style={{ width: "100px", height: "100px", borderRadius: "50%" }}
          />
          <div style={{ marginTop: "8px" }}>{playerObj.player.name}</div>
        </div>
      ))}
    </div>
  );
}

function App() {
  return (
    <div>
      <h1>Premier League Top Scorers</h1>
      <TopScorers />
    </div>
  );
}

export default App;
